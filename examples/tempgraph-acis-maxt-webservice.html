tempgraph-acis-maxt-webservice.xml

<mugl>
  <background color="0xcccccc"/>
    <window margin="0" padding="5" border="5" bordercolor="0x000000"/>
    <plotarea margintop="10" marginleft="38" marginbottom="35" marginright="35"/>
    <horizontalaxis id="x" type="datetime" min="20120601" max="20120608" length="1" base="-1,-1" anchor="-1" tickmin="-5" tickmax="5">
      <labels format="%n %d" start="0" spacing="7D 2D 1D" anchor="0,0" position="10,-10" angle="45" />
    </horizontalaxis>
    <verticalaxis   id="y"  min="20" max="100" length="1">
      <labels format="%.2f" start="0" spacing="100 20 10 5 2 1 0.5 0.2 0.1 0.05 0.02 0.01" anchor="0,0" position="0,0" angle="0" />
    </verticalaxis>
    <plot>
      <horizontalaxis ref="x">
        <variable ref="datetime"/>
      </horizontalaxis>
      <verticalaxis ref="y">
        <variable ref="maxt"/>
      </verticalaxis>
      <renderer type="pointline">
        <option name="linecolor" value="0xff00ff"/>
        <option name="linewidth" value="3"/>
        <option name="pointshape" value="circle"/>
        <option name="pointoutlinewidth" value="2"/>
        <option name="pointoutlinecolor" value="0x000000"/>
        <option name="pointcolor" value="0xffffff"/>
        <option name="pointopacity" value="1.0"/>
        <option name="pointsize" value="3"/>
      </renderer>
    </plot>
    <data>
      <variables>
        <variable column="0" type="datetime" id="datetime"/>
        <variable column="1" type="number" id="maxt" missingvalue="-9000" missingop="le"/>
      </variables>
      <service location="http://acis.multigraph.org/acisdata/13881-1/maxt/" format="%Y%M%D"/>
    </data>
</mugl>
